  <script>
    // Step 1: A function that simulates a delay (like fetching data from server)
    function fetchData(callback) {
      document.getElementById("output").innerHTML = "Fetching data... Please wait ‚è≥";

      // setTimeout() is asynchronous ‚Äì it waits 2 seconds before running the code inside
      setTimeout(function() {
        let data = "üê± Cat data loaded successfully!";
        callback(data); // calling the callback function
      }, 2000);
    }

    // Step 2: The callback function
    function displayData(result) {
      document.getElementById("output").innerHTML = result;
    }

    // Step 3: Calling fetchData and passing displayData as a callback
    fetchData(displayData);

    // This message prints immediately (shows asynchronous behavior)
    console.log("This message appears first because fetchData is asynchronous!");
  </script>
